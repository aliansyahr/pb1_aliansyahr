function tampilkanAngka() {
  var list = document.getElementById("numbersList");
  for (var i = 1; i <= 10; i++) {
    var listItem = document.createElement("li");

    if (i % 2 === 0) {
      listItem.textContent = "Even: " + i;
    } else {
      listItem.textContent = "Odd: " + i;
    }
      list.appendChild(listItem);
    }
  }
